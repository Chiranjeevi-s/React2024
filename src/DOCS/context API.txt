create your context on top of the Parent component where you import 
Eg: 
import { createContext } from "react";
const MyContext = createContext<any>(null);
like this

In you Child component you can use like this
      <MyContext.Consumer>
        {(name) => {
          return (
            <>
              {name.map((item: any, index: any) => (
                <div key={index}>
                  Hii I'am {item.name} am {item.sex} and am {item.age} old
                </div>
              ))}
            </>
          );
        }}
      </MyContext.Consumer>
